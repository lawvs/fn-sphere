Filter Sphere provides a set of themes that you can use to customize the look and feel of the Filter Sphere.

## Preset Themes

import PackageManagerTabs from "/src/components/package-manager-tabs.astro";

### Material UI

The Material UI theme provides a sleek, modern look for your Filter Sphere components, based on [MUI Material UI](https://mui.com/material-ui/).

<PackageManagerTabs package="@fn-sphere/theme-mui-material" />

```tsx live  wrapper="/src/components/mui-theme-wrapper.tsx" {2} {13}
// 1. Import the MUI Material theme
import { filterTheme } from "@fn-sphere/theme-mui-material";
import {
  FilterSphereProvider,
  FilterBuilder,
  useFilterSphere,
} from "@fn-sphere/filter";

export default function AdvancedFilter({ schema }) {
  const { predicate, context } = useFilterSphere({ schema });
  return (
    // 2. Set the theme for the Filter Sphere
    <FilterSphereProvider context={context} theme={filterTheme}>
      <FilterBuilder />
    </FilterSphereProvider>
  );
}
```

## Overriding the theme

You can override Filter Sphere's built-in components by replacing them with your own custom components or templates within the theme.

The following example shows how to override the Filter Sphere's build-in theme to use a custom component.

```tsx live wrapper="/src/components/mui-theme-wrapper.tsx"
import { filterTheme } from "@fn-sphere/theme-mui-material";
import { Button } from "@mui/material";
import {
  FilterSphereProvider,
  FilterBuilder,
  useFilterSphere,
  useFilterRule,
} from "@fn-sphere/filter";

const customTheme = {
  ...filterTheme,
  components: {
    ...filterTheme.components,
    Button: (props) => (
      <Button variant="outlined" color="success" size="small" {...props} />
    ),
  },
  templates: {
    ...filterTheme.templates,
    SingleFilter: (props) => {
      const MuiSingleFilter = filterTheme.templates.SingleFilter;
      const {
        ruleState: { isFirstRule, parentGroup },
      } = useFilterRule(props.rule);
      return (
        <div style={{ display: "flex", alignItems: "center" }}>
          {isFirstRule && <span style={{ marginRight: "8px" }}>Where</span>}
          {!isFirstRule && (
            <span style={{ marginRight: "8px" }}>{parentGroup.op}</span>
          )}
          <MuiSingleFilter {...props} />
        </div>
      );
    },
  },
};

export default function AdvancedFilter({ schema }) {
  const { predicate, context } = useFilterSphere({ schema });
  return (
    <FilterSphereProvider context={context} theme={customTheme}>
      <FilterBuilder />
    </FilterSphereProvider>
  );
}
```
